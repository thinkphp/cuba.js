How it works


cuba.ready(function() {

         //select an element and added a handler on event 'click'
         cuba.select("#btn").on('click', function(){

              cuba.select('#out').toggleClass('myClass')

              alert('clicked')
         })

         //added label
         cuba.select("#btn").html("Click Me")  

         //select an element, then called the methods: html and css
         cuba.select("#out")
             .html("Jean Baptiste Poquelin")
             .css("width: 100px; height: 100px;border: 1px solid #ccc;background: #393;color: #fff")

         //make an YQL request
         cuba.yql("select title from rss where url=\"http://rss.news.yahoo.com/rss/topstories\"", function( data ) {
                   if(window.console) console.log( data ) 
         })

         //cuba.addClass($('out'),'myClass') 
              //OR
         cuba.select('#out').addClass('myClass')


         //using CSS3 transitions
         $('move').Click( function( event ){

                 cuba.stopPropagation( event ) 
 
                 cuba.select('#sandbox .box').css('margin-left:100px') 
         } )

         $('moveback').Click( function( event ){

                 cuba.stopPropagation( event ) 
 
                 cuba.select('#sandbox .box').css('margin-left:0px') 
         } )

})
