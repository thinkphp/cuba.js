cuba.ready(function(){
 
         cuba.delegate($('#menu'), 'li a', 'click', function( e ){

              var ul = e.target.parentNode.querySelector("ul")

                  if(ul.style.display == "none") {

                     ul.style.display = "block"

                  } else {

                    ul.style.display = "none"
                  }

                  console.log(e.target)

         }, false);
})
