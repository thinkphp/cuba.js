Autosuggest.js

The autosuggest widget provides suggestions while you type into the field. The datasource is a simple JS array, provided to the widget using the second @param `suggestions`.

Syntax:

cuba.UI.autosuggest ( elem, suggestions )



var $ = function( id ) {return document.getElementById( id ) }

var suggestions = ["jquery","mootools","dojo","nodejs","angular","yui3","yui2"]

var newS = suggestions.concat(["express"])

var suggest = cuba.UI.autosuggest($('input'), newS)

    suggest.set(newS.concat(["prototype.js"]))
